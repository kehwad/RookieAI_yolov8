## 项目概述

RookieAI_yolov8 是一个基于 YOLOv8 的游戏目标检测自瞄助手，原本专为 Apex Legends 设计，现在支持训练适配其他 FPS 游戏的自定义模型。

### 核心特性
- 🎯 基于 YOLOv8 的实时目标检测
- 🚀 支持多种模型格式（.pt/.engine/.onnx）
- 🎮 多进程架构，高性能推理
- 🖱️ 多种鼠标控制方式（win32/mouse/罗技/飞易来）
- ⚙️ 丰富的参数配置选项

### 支持的游戏类型
- FPS 射击游戏（CS2、VALORANT、Apex Legends、Call of Duty 等）
- 战术射击游戏
- 其他需要目标识别的竞技游戏

## 数据收集

### 1. 游戏截图收集策略

#### 自动截图脚本
创建一个自动截图脚本来收集游戏画面：

```python
# scripts/data_collection.py
import cv2
import numpy as np
import mss
import time
import os
from datetime import datetime
import keyboard

def collect_game_screenshots(game_name, duration_minutes=60):
    """
    自动收集游戏截图
    
    Args:
        game_name: 游戏名称
        duration_minutes: 收集时长（分钟）
    """
    
    # 创建数据目录
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    data_dir = f"training_data/{game_name}_{timestamp}"
    os.makedirs(data_dir, exist_ok=True)
    
    print(f"开始收集 {game_name} 的训练数据...")
    print("按下 'q' 键停止收集")
    
    with mss.mss() as sct:
        # 获取主显示器
        monitor = sct.monitors[1]
        
        start_time = time.time()
        frame_count = 0
        
        while True:
            # 检查是否超时或按键停止
            if keyboard.is_pressed('q'):
                break
            if (time.time() - start_time) > (duration_minutes * 60):
                break
                
            # 截取屏幕
            screenshot = sct.grab(monitor)
            img = np.array(screenshot)
            img = cv2.cvtColor(img, cv2.COLOR_BGRA2RGB)
            
            # 保存图像
            filename = f"{data_dir}/frame_{frame_count:06d}.jpg"
            cv2.imwrite(filename, cv2.cvtColor(img, cv2.COLOR_RGB2BGR))
            
            frame_count += 1
            time.sleep(0.1)  # 每秒10帧
            
            if frame_count % 100 == 0:
                print(f"已收集 {frame_count} 张图片...")
    
    print(f"数据收集完成！共收集 {frame_count} 张图片")
    print(f"数据保存在: {data_dir}")

if __name__ == "__main__":
    # 使用示例
    collect_game_screenshots("CS2", duration_minutes=30)
```

#### 数据收集建议
1. **多样性场景**：
   - 不同地图/场景
   - 不同时间段（白天/夜晚）
   - 不同武器视角
   - 不同距离的敌人

2. **数据质量**：
   - 分辨率保持一致（建议1920x1080）
   - 避免过度模糊的图片
   - 确保目标清晰可见
   - 包含各种姿态的敌人（站立/蹲伏/趴下）

3. **数据量建议**：
   - 最少：1000张高质量图片
   - 推荐：3000-5000张图片
   - 理想：10000+张图片

```python
# scripts/annotation_helper.py
import os
import xml.etree.ElementTree as ET
from pathlib import Path

def validate_annotations(annotations_dir):
    """
    验证标注文件的有效性
    """
    valid_count = 0
    error_files = []
    
    for xml_file in Path(annotations_dir).glob("*.xml"):
        try:
            tree = ET.parse(xml_file)
            root = tree.getroot()
            
            # 检查基本结构
            if root.find('filename') is None:
                error_files.append(f"{xml_file}: 缺少filename标签")
                continue
                
            # 检查边界框
            objects = root.findall('object')
            if len(objects) == 0:
                error_files.append(f"{xml_file}: 没有标注对象")
                continue
                
            valid_count += 1
            
        except Exception as e:
            error_files.append(f"{xml_file}: {str(e)}")
    
    print(f"有效标注文件: {valid_count}")
    print(f"错误文件: {len(error_files)}")
    for error in error_files:
        print(f"  - {error}")

def convert_to_yolo_format(xml_dir, output_dir, classes):
    """
    将标注文件转换为YOLO格式
    """
    os.makedirs(output_dir, exist_ok=True)
    
    for xml_file in Path(xml_dir).glob("*.xml"):
        tree = ET.parse(xml_file)
        root = tree.getroot()
        
        # 获取图片尺寸
        width = int(root.find('size/width').text)
        height = int(root.find('size/height').text)
        
        # 创建对应的txt文件
        txt_file = output_dir / f"{xml_file.stem}.txt"
        
        with open(txt_file, 'w') as f:
            for obj in root.findall('object'):
                class_name = obj.find('name').text
                if class_name not in classes:
                    continue
                    
                class_id = classes[class_name]
                
                # 获取边界框坐标
                xmin = int(obj.find('bndbox/xmin').text)
                ymin = int(obj.find('bndbox/ymin').text)
                xmax = int(obj.find('bndbox/xmax').text)
                ymax = int(obj.find('bndbox/ymax').text)
                
                # 转换为YOLO格式 (归一化)
                x_center = (xmin + xmax) / 2.0 / width
                y_center = (ymin + ymax) / 2.0 / height
                w = (xmax - xmin) / width
                h = (ymax - ymin) / height
                
                f.write(f"{class_id} {x_center:.6f} {y_center:.6f} {w:.6f} {h:.6f}\n")
```

```python
# scripts/train_model.py
from ultralytics import YOLO
import wandb
import torch

def train_game_model(game_name, epochs=300, batch_size=16, img_size=640):
    """
    训练游戏专用模型
    
    Args:
        game_name: 游戏名称
        epochs: 训练轮数
        batch_size: 批处理大小
        img_size: 输入图片尺寸
    """
    
    # 初始化 wandb（可选，用于训练可视化）
    # wandb.init(project=f"rookieai-{game_name}", name=f"{game_name}_yolov8n")
    
    # 加载预训练模型
    model = YOLO('yolov8n.pt')  # 可选: yolov8s.pt, yolov8m.pt, yolov8l.pt
    
    # 检查GPU可用性
    device = 'cuda' if torch.cuda.is_available() else 'cpu'
    print(f"使用设备: {device}")
    
    # 训练参数配置
    results = model.train(
        data='dataset/data.yaml',      # 数据集配置文件
        epochs=epochs,                 # 训练轮数
        batch=batch_size,             # 批处理大小
        imgsz=img_size,               # 输入图片尺寸
        device=device,                # 训练设备
        workers=4,                    # 数据加载线程数
        project=f'runs/{game_name}',  # 项目保存路径
        name='train',                 # 实验名称
        
        # 数据增强参数
        hsv_h=0.015,      # 色调增强
        hsv_s=0.7,        # 饱和度增强
        hsv_v=0.4,        # 亮度增强
        degrees=0.0,      # 旋转角度
        translate=0.1,    # 平移
        scale=0.5,        # 缩放
        shear=0.0,        # 剪切
        perspective=0.0,  # 透视变换
        flipud=0.0,       # 垂直翻转
        fliplr=0.5,       # 水平翻转
        mosaic=1.0,       # 马赛克增强
        mixup=0.0,        # 混合增强
        
        # 优化器参数
        optimizer='SGD',   # 优化器 (SGD, Adam, AdamW)
        lr0=0.01,         # 初始学习率
        lrf=0.1,          # 最终学习率比例
        momentum=0.937,   # SGD动量
        weight_decay=0.0005,  # 权重衰减
        warmup_epochs=3,  # 预热轮数
        warmup_momentum=0.8,  # 预热动量
        warmup_bias_lr=0.1,   # 预热偏置学习率
        
        # 验证参数
        val=True,         # 是否进行验证
        save=True,        # 是否保存检查点
        save_period=50,   # 保存检查点间隔
        cache=False,      # 是否缓存图片到内存
        rect=False,       # 矩形训练
        cos_lr=True,      # 余弦学习率调度
        close_mosaic=10,  # 关闭mosaic的最后轮数
        resume=False,     # 是否恢复训练
        amp=True,         # 自动混合精度
        fraction=1.0,     # 使用数据集的比例
        profile=False,    # 性能分析
        
        # 多尺度训练
        multi_scale=True,
        
        # 早停参数
        patience=100,     # 早停耐心值
    )
    
    # 保存最终模型
    model_path = f'models/{game_name}_yolov8n.pt'
    model.save(model_path)
    
    print(f"训练完成！模型已保存至: {model_path}")
    print(f"训练结果保存在: runs/{game_name}/train/")
    
    return results

if __name__ == "__main__":
```

## 模型优化

### 1. 模型格式转换

将训练好的.pt模型转换为更高效的格式：

```python
# scripts/optimize_model.py
from ultralytics import YOLO
import torch

def optimize_model(model_path, export_format='engine'):
    """
    优化模型以提升推理速度
    
    Args:
        model_path: 原始模型路径
        export_format: 导出格式 ('engine', 'onnx', 'torchscript')
    """
    
    # 加载训练好的模型
    model = YOLO(model_path)
    
    if export_format == 'engine':
        # 转换为 TensorRT 格式（需要NVIDIA GPU）
        print("正在转换为 TensorRT 格式...")
        model.export(
            format='engine',
            workspace=8.0,      # GPU内存工作空间(GB)
            half=True,          # 使用FP16精度
            dynamic=False,      # 静态输入尺寸
            simplify=True,      # 简化模型
            opset=11,          # ONNX opset版本
        )
        
    elif export_format == 'onnx':
        # 转换为 ONNX 格式
        print("正在转换为 ONNX 格式...")
        model.export(
            format='onnx',
            half=True,
            dynamic=False,
            simplify=True,
            opset=11,
        )
        
    elif export_format == 'torchscript':
        # 转换为 TorchScript 格式
        print("正在转换为 TorchScript 格式...")
        model.export(format='torchscript')
    
    print(f"模型优化完成！格式: {export_format}")

def benchmark_model(model_path):
    """
    测试模型性能
    """
    import time
    import numpy as np
    
    model = YOLO(model_path)
    
    # 创建测试图像
    test_img = np.random.randint(0, 255, (640, 640, 3), dtype=np.uint8)
    
    # 预热
    for _ in range(10):
        model.predict(test_img, verbose=False)
    
    # 性能测试
    times = []
    for _ in range(100):
        start = time.time()
        results = model.predict(test_img, verbose=False)
        end = time.time()
        times.append(end - start)
    
    avg_time = np.mean(times) * 1000  # 转换为毫秒
    fps = 1000 / avg_time
    
    print(f"平均推理时间: {avg_time:.2f}ms")
    print(f"推理帧率: {fps:.2f} FPS")
    
    return avg_time, fps

if __name__ == "__main__":
    # 优化模型
    optimize_model("models/CS2_yolov8n.pt", export_format='engine')
    
    # 性能测试
    benchmark_model("models/CS2_yolov8n.engine")
```

## 常见问题

### 1. 训练相关问题

**Q: 训练过程中出现 CUDA out of memory 错误？**
A: 
- 减小 batch_size（如从16减到8或4）
- 减小图片尺寸（如从640减到320）
- 使用梯度累积：设置 `accumulate=True`

**Q: 模型训练不收敛，loss不下降？**
A:
- 检查数据标注质量
- 降低学习率：`lr0=0.001`
- 增加训练轮数：`epochs=500`
- 使用预训练模型：确保使用 `yolov8n.pt` 而不是从头训练

**Q: 验证精度很低？**
A:
- 检查验证集标注是否正确
- 确保训练集和验证集的数据分布一致
- 增加数据增强
- 检查类别平衡性

### 2. 部署相关问题

**Q: 模型推理速度太慢？**
A:
- 转换为 TensorRT 格式：`.engine`
- 使用更小的模型：`yolov8n` 而不是 `yolov8l`
- 启用 FP16：`half=True`
- 减小输入图片尺寸

**Q: 检测精度不够？**
A:
- 调整置信度阈值
- 重新标注训练数据
- 增加训练数据量
- 使用更大的模型

**Q: 鼠标移动不准确？**
A:
- 校准鼠标灵敏度参数
- 调整瞄准偏移量
- 检查游戏内鼠标设置
- 关闭鼠标加速

### 3. 兼容性问题

**Q: 某些游戏无法使用？**
A:
- 尝试不同的鼠标移动方式
- 检查反作弊软件限制
- 使用 KmBoxNet 等硬件方案
- 调整检测区域

```python
# 针对不同目标类别使用不同策略
def advanced_target_selection(detections, target_priorities):
    """
    高级目标选择算法
    """
    if not detections:
        return None
    
    best_target = None
    best_score = 0
    
    for detection in detections:
        class_id = int(detection.cls)
        confidence = float(detection.conf)
        
        # 计算距离权重
        x_center = (detection.xyxy[0] + detection.xyxy[2]) / 2
        y_center = (detection.xyxy[1] + detection.xyxy[3]) / 2
        distance = np.sqrt((x_center - 960)**2 + (y_center - 540)**2)
        distance_weight = 1.0 / (1.0 + distance / 100)
        
        # 计算目标大小权重
        width = detection.xyxy[2] - detection.xyxy[0]
        height = detection.xyxy[3] - detection.xyxy[1]
        size_weight = np.sqrt(width * height) / 10000
        
        # 综合评分
        priority = target_priorities.get(class_id, 1.0)
        total_score = confidence * distance_weight * size_weight * priority
        
        if total_score > best_score:
            best_score = total_score
            best_target = detection
    
    return best_target
```

```python
# 实时性能监控
class PerformanceMonitor:
    def __init__(self):
        self.fps_history = []
        self.accuracy_history = []
        
    def log_performance(self, fps, accuracy):
        self.fps_history.append(fps)
        self.accuracy_history.append(accuracy)
        
        # 保持历史记录在合理范围内
        if len(self.fps_history) > 1000:
            self.fps_history.pop(0)
            self.accuracy_history.pop(0)
    
    def get_average_performance(self):
        if not self.fps_history:
            return 0, 0
        return np.mean(self.fps_history), np.mean(self.accuracy_history)
    
    def performance_alert(self):
        avg_fps, avg_acc = self.get_average_performance()
        
        if avg_fps < 30:
            print("警告：帧率过低，考虑优化模型或降低图像质量")
        if avg_acc < 0.5:
            print("警告：检测精度较低，考虑重新训练模型")
```
